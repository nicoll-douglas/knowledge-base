\newglossaryentry{packetswitching}{
    name={Packet Switching},
    description={A network transmission method that breaks data up into into small chunks called packets before transmission and is then reassembled at the destination}
}

\newglossaryentry{storeandforward}{
    name={Store-and-forward},
    description={A transmission method where packet switches must receive an entire packet and process it whole before it can transmit the first bit of the packet onto the outbound link}
}
